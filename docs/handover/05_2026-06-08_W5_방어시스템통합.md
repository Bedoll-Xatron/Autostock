```python
# ── 섹터 + 보유 한도 검증 ──────────────────
existing_holdings = db.fetch_held_positions()
existing_sectors = [_lookup_sector(h["ticker"]) for h in existing_holdings]
sector_count = {s: existing_sectors.count(s) for s in set(existing_sectors)}
total_holding = len(existing_holdings)
new_count = 0

for d in decisions:
    if d.action != "BUY":
        continue

    sec = w_map.get(d.ticker, {}).get("sector", "Unknown")

    # 섹터 캡
    if sector_count.get(sec, 0) >= SECTOR_LIMIT:
        log.info("[%s] 섹터 %s 한도(%d) 초과 — 보류", d.ticker, sec, SECTOR_LIMIT)
        # blocked_reason="SECTOR_LIMIT" 으로 기록
        continue

    # 일일 한도
    if new_count >= DAILY_NEW_ENTRY_LIMIT:
        log.info("[%s] 일일 신규 진입 한도(%d) 초과 — 보류", d.ticker, DAILY_NEW_ENTRY_LIMIT)
        continue

    # 총 보유 한도
    if total_holding + new_count >= TOTAL_HOLDING_LIMIT:
        log.info("[%s] 총 보유 한도(%d) 초과 — 보류", d.ticker, TOTAL_HOLDING_LIMIT)
        continue

    sector_count[sec] = sector_count.get(sec, 0) + 1
    new_count += 1
    # → 매수 진행
```

### Step 2. 드로다운 서킷 브레이커 (월)

**파일 (신규):** `autostock/trading/circuit_breaker.py`

```python
"""포트폴리오 누적 손실 서킷 브레이커."""
from datetime import date, timedelta
from autostock.db import supabase as db
from autostock.logger import get_logger

log = get_logger(__name__)

CIRCUIT_DRAWDOWN_PCT = -5.0    # 최근 5일 누적 -5% 이하면 발동
CIRCUIT_LOOKBACK_DAYS = 5
CIRCUIT_PAUSE_DAYS = 7         # 발동 시 일주일 매수 중단


def check_circuit_breaker() -> tuple[bool, str]:
    """
    Returns:
        (paused, reason): paused=True면 매수 중단
    """
    # 최근 5일 손익 (decisions 테이블의 BUY → 후속 매도 결과)
    end = date.today()
    start = end - timedelta(days=CIRCUIT_LOOKBACK_DAYS)

    rows = db.fetch_pnl_between(start.isoformat(), end.isoformat())
    if not rows:
        return False, "no_data"

    cumulative_pct = sum(r["pnl_pct"] for r in rows) / max(len(rows), 1)
    if cumulative_pct <= CIRCUIT_DRAWDOWN_PCT:
        log.warning("서킷 브레이커 발동: %d일 누적 %.2f%% <= %.1f%%",
                    CIRCUIT_LOOKBACK_DAYS, cumulative_pct, CIRCUIT_DRAWDOWN_PCT)
        return True, f"누적 {cumulative_pct:+.1f}% (최근 {CIRCUIT_LOOKBACK_DAYS}일)"

    # 최근 발동 후 PAUSE_DAYS 미경과 시 유지
    last_trip = db.fetch_last_circuit_trip()
    if last_trip:
        days_since = (end - date.fromisoformat(last_trip)).days
        if days_since < CIRCUIT_PAUSE_DAYS:
            return True, f"서킷 일시정지 ({CIRCUIT_PAUSE_DAYS - days_since}일 남음)"

    return False, "ok"
```

**파일:** `autostock/scheduler/jobs.py:run_daily_pipeline`

거시 필터 다음에 추가:
```python
# ── 서킷 브레이커 ──
from autostock.trading.circuit_breaker import check_circuit_breaker
paused, reason = check_circuit_breaker()
if paused:
    log.warning("서킷 브레이커 발동 — 신규 매수 중단: %s", reason)
    bot_ui.schedule_message(f"🚦 *서킷 브레이커* 발동 — {reason}\n신규 매수 중단")
    return

### Step 4. RVOL 게이트화 (W4 보강) (화)

    # ... 기존 점수 계산 ...

    # mandatory: 거래량 검증 통과 필수

        score.mandatory_passed = False  # 이미 false 였으면 그대로
    # 기존 mandatory_passed 로직 유지

    return score, checklist
```
